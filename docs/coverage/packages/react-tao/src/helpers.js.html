export function normalizeAC({ t, term, a, action, o, orient }) {
  return {
    term: term || t,

export const cleanInput = ({ term, action, orient }) => {
  const incoming = { term, action, orient };
  Object.keys(incoming).forEach(k => incoming[k] == null && delete incoming[k]);
  return incoming;
};
